428  unpack(std::map<Sym<REAL>, ParticleDatSharedPtr<REAL>> &particle_dats_real,
429  std::map<Sym<INT>, ParticleDatSharedPtr<INT>> &particle_dats_int) {
430 
431  auto t0 = profile_timestamp();
432 
433  // copy packed data to device
434 
435  const int cpysize = this->npart_recv * this->num_bytes_per_particle;
436  sycl::event event_memcpy;
437  if (cpysize > 0) {
438  event_memcpy = this->sycl_target->queue.memcpy(
439  this->d_recv_buffer.ptr, this->h_recv_buffer.ptr, cpysize);
440  }
441 
442  // old cell occupancy
443  auto mpi_rank_dat = particle_dats_int[Sym<INT>("NESO_MPI_RANK")];
444  const int npart_cell_0_old = mpi_rank_dat->h_npart_cell[0];
445  const int npart_cell_0_new = npart_cell_0_old + this->npart_recv;
446  // realloc cell 0 on the dats
447  for (auto &dat : particle_dats_real) {
448  dat.second->realloc(0, npart_cell_0_new);
449  dat.second->set_npart_cell(0, npart_cell_0_new);
450  }
451  for (auto &dat : particle_dats_int) {
452  dat.second->realloc(0, npart_cell_0_new);
453  dat.second->set_npart_cell(0, npart_cell_0_new);
454  }
455  for (auto &dat : particle_dats_real) {
456  dat.second->wait_realloc();
457  }
458  for (auto &dat : particle_dats_int) {
459  dat.second->wait_realloc();
460  }
461 
462  const int k_npart_recv = this->npart_recv;
463 
464  // get the pointers to the particle dat data and the number of components in
465  // each dat
466  get_particle_dat_info(particle_dats_real, particle_dats_int);
467 
468  // unpack into cell 0
469  const int k_num_bytes_per_particle = this->num_bytes_per_particle;
470  const int k_num_dats_real = this->num_dats_real;
471  const int k_num_dats_int = this->num_dats_int;
472  const auto k_particle_dat_ptr_real =
473  this->dh_particle_dat_ptr_real.d_buffer.ptr;
474  const auto k_particle_dat_ptr_int =
475  this->dh_particle_dat_ptr_int.d_buffer.ptr;
476  const auto k_particle_dat_ncomp_real =
477  this->dh_particle_dat_ncomp_real.d_buffer.ptr;
478  const auto k_particle_dat_ncomp_int =
479  this->dh_particle_dat_ncomp_int.d_buffer.ptr;
480  char *k_recv_buffer = this->d_recv_buffer.ptr;
481 
482  sycl_target->profile_map.inc("ParticleUnpacker", "unpack_prepare", 1,
483  profile_elapsed(t0, profile_timestamp()));
484 
485  if (cpysize > 0) {
486  event_memcpy.wait_and_throw();
487  }
488 
489  this->sycl_target->queue
490  .submit([&](sycl::handler &cgh) {
491  cgh.parallel_for<>(
492  // for each new particle
493  sycl::range<1>(static_cast<size_t>(k_npart_recv)),
494  [=](sycl::id<1> idx) {
495  const int cell = 0;
496  // destination layer in the cell
497  const int layer = npart_cell_0_old + idx;
498  // source position in the packed buffer
499  const int offset = k_num_bytes_per_particle * idx;
500  char *unpack_base_ptr = k_recv_buffer + offset;
501  REAL *unpack_ptr_real = (REAL *)unpack_base_ptr;
502  // for each real dat
503  int index = 0;
504  for (int dx = 0; dx < k_num_dats_real; dx++) {
505  REAL ***dat_ptr = k_particle_dat_ptr_real[dx];
506  const int ncomp = k_particle_dat_ncomp_real[dx];
507  // for each component
508  for (int cx = 0; cx < ncomp; cx++) {
509  dat_ptr[cell][cx][layer] = unpack_ptr_real[index + cx];
510  }
511  index += ncomp;
512  }
513  // for each int dat
514  INT *unpack_ptr_int = (INT *)(unpack_ptr_real + index);
515  index = 0;
516  for (int dx = 0; dx < k_num_dats_int; dx++) {
517  INT ***dat_ptr = k_particle_dat_ptr_int[dx];
518  const int ncomp = k_particle_dat_ncomp_int[dx];
519  // for each component

522  }
523  index += ncomp;
524  }
525  });
526  })
527  .wait_and_throw();
528  }
