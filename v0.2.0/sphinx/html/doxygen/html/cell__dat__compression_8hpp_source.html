106  template <typename T>
107  inline void compute_remove_compress_indicies(const int npart, T *usm_cells,
108  T *usm_layers) {
109  auto t0 = profile_timestamp();
110 
111  d_compress_cells_old.realloc_no_copy(npart);
112  d_compress_layers_old.realloc_no_copy(npart);
113  d_compress_layers_new.realloc_no_copy(npart);
114 
115  NESOASSERT(this->d_npart_cell.size >= this->ncell,
116  "Bad device_npart_cell length");
117 
118  const int ncell = this->ncell;
119 
120  auto mpi_particle_dat = this->particle_dats_int[Sym<INT>("NESO_MPI_RANK")];
121  auto k_npart_cell = mpi_particle_dat->d_npart_cell;
122 
123  auto device_npart_cell_ptr = this->d_npart_cell.ptr;
124  auto device_move_counters_ptr = this->d_move_counters.ptr;
125 
126  auto compress_cells_old_ptr = this->d_compress_cells_old.ptr;
127  auto compress_layers_old_ptr = this->d_compress_layers_old.ptr;
128  auto compress_layers_new_ptr = this->d_compress_layers_new.ptr;
129 
130  INT ***cell_ids_ptr = this->cell_id_dat->cell_dat.device_ptr();
131  this->sycl_target->queue
132  .submit([&](sycl::handler &cgh) {
133  cgh.parallel_for<>(sycl::range<1>(static_cast<size_t>(ncell)),
134  [=](sycl::id<1> idx) {
135  device_npart_cell_ptr[idx] =
136  static_cast<int>(k_npart_cell[idx]);
137  device_move_counters_ptr[idx] = 0;
138  });
139  })
140  .wait();
141 
142  this->sycl_target->queue
143  .submit([&](sycl::handler &cgh) {
144  cgh.parallel_for<>(
145  sycl::range<1>(static_cast<size_t>(npart)), [=](sycl::id<1> idx) {
146  const auto cell = usm_cells[idx];
147  const auto layer = usm_layers[idx];
148 
149  // Atomically do device_npart_cell_ptr[cell]--
150  sycl::atomic_ref<int, sycl::memory_order::relaxed,
151  sycl::memory_scope::device>
152  element_atomic(device_npart_cell_ptr[cell]);
153  auto count = element_atomic.fetch_add(-1);
154 
157  cell_ids_ptr[cell][0][layer] = -1;
158  });
159  })
160  .wait();
161 
162  this->sycl_target->queue
163  .submit([&](sycl::handler &cgh) {
164  cgh.parallel_for<>(
165  sycl::range<1>(static_cast<size_t>(npart)), [=](sycl::id<1> idx) {
166  const auto cell = usm_cells[idx];
167  const auto layer = usm_layers[idx];
168 
169  // Is this layer less than the new cell count?
170  // If so then there is a particle in a row greater than the cell
171  // count to be copied into this layer.
172  if (layer < device_npart_cell_ptr[cell]) {
173 
174  // If there are n rows to be filled in row indices less than
175  // the new cell count then there are n rows greater than the
176  // new cell count which are to be copied down. Atomically
177  // compute which one of those rows this index copies.
178  sycl::atomic_ref<int, sycl::memory_order::relaxed,
179  sycl::memory_scope::device>
180  element_atomic(device_move_counters_ptr[cell]);
181  const INT source_row_offset = element_atomic.fetch_add(1);
182 
183  // find the source row counting up from new_cell_count to
184  // old_cell_count potential source rows will have a cell index
185  // >= 0. This index should pick the i^th source row where i
186  // was computed from the atomic above.
187  INT found_count = 0;
188  INT source_row = -1;
189  for (INT rowx = device_npart_cell_ptr[cell];
190  rowx < k_npart_cell[cell]; rowx++) {
191  // Is this a potential source row?
192  if (cell_ids_ptr[cell][0][rowx] > -1) {
193  if (source_row_offset == found_count++) {
194  source_row = rowx;
195  break;
196  }
197  }
198  }
199  compress_cells_old_ptr[idx] = cell;
200  compress_layers_new_ptr[idx] = layer;
201  compress_layers_old_ptr[idx] = source_row;
202 
203  } else {
204 
205  compress_cells_old_ptr[idx] = -1;
206  compress_layers_new_ptr[idx] = -1;
207  compress_layers_old_ptr[idx] = -1;
208  }
209  });
210  })
211  .wait_and_throw();
212  sycl_target->profile_map.inc("LayerCompressor",
213  "compute_remove_compress_indicies", 1,
214  profile_elapsed(t0, profile_timestamp()));
215  }

241 
242  for (auto &dat : particle_dats_real) {

244  npart, compress_cells_old_ptr, compress_cells_old_ptr,
245  compress_layers_old_ptr, compress_layers_new_ptr));
246  this->event_stack.push(
247  dat.second->set_npart_cells_device(this->d_npart_cell.ptr));
248  }

292  profile_elapsed(t4, profile_timestamp()));
293 
294  sycl_target->profile_map.inc("LayerCompressor", "remove_particles", 1,
295  profile_elapsed(t0, profile_timestamp()));
296  }
