287  inline void set_nrow(const INT cell, const INT nrow_required) {
288 
289  auto t0 = profile_timestamp();
290 
291  NESOASSERT(cell >= 0, "Cell index is negative");
292  NESOASSERT(cell < this->ncells, "Cell index is >= ncells");
293  NESOASSERT(nrow_required >= 0, "Requested number of rows is negative");
294 
295  if (nrow_required < this->nrow[cell]) {
296  this->nrow[cell] = nrow_required;
297  this->nrow_max = -1;
298  sycl_target->profile_map.inc("CellDat", "set_nrow", 1,
299  profile_elapsed(t0, profile_timestamp()));
300  return;
301  }
302 
303  const INT nrow_alloced = this->nrow_alloc[cell];
304  const INT nrow_existing = this->nrow[cell];
305 
306  if (nrow_required != nrow_existing) {
307  if (nrow_required > nrow_alloced) {
308  for (int colx = 0; colx < this->ncol; colx++) {
309  T *col_ptr_old = this->h_ptr_cols[cell * this->ncol + colx];
310  this->stack_ptrs.push(col_ptr_old);
311  T *col_ptr_new =
312  (T *)this->sycl_target->malloc_device(nrow_required * sizeof(T));
313  NESOASSERT(col_ptr_new != nullptr, "bad pointer from malloc_device");
314 
315  if (nrow_alloced > 0) {
316  this->stack_events.push(this->sycl_target->queue.memcpy(
317  col_ptr_new, col_ptr_old, nrow_existing * sizeof(T)));
318  }
319  this->h_ptr_cols[cell * this->ncol + colx] = col_ptr_new;
320  }
321  this->nrow_alloc[cell] = nrow_required;
322  this->stack_events.push(sycl_target->queue.memcpy(
323  this->h_ptr_cells[cell], &this->h_ptr_cols[cell * this->ncol],
324  this->ncol * sizeof(T *)));
325  }
326  this->nrow[cell] = nrow_required;
327  this->nrow_max = -1;
328  }
329  sycl_target->profile_map.inc("CellDat", "set_nrow", 1,
330  profile_elapsed(t0, profile_timestamp()));
331  }
