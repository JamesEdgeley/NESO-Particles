105  cgh.single_task<>([=]() { k_lookup_count[0] = 0; });

115  auto k_position_dat = this->position_dat->cell_dat.device_ptr();
116  auto k_mpi_rank_dat = this->mpi_rank_dat->cell_dat.device_ptr();
117 
118  // iteration set
119  auto pl_iter_range = this->mpi_rank_dat->get_particle_loop_iter_range();
120  auto pl_stride = this->mpi_rank_dat->get_particle_loop_cell_stride();
121  auto pl_npart_cell = this->mpi_rank_dat->get_particle_loop_npart_cell();
122 
123  // pointers to access BufferDevices in the kernel
124  auto k_lookup_local_cells = this->d_lookup_local_cells.ptr;
125  auto k_lookup_local_layers = this->d_lookup_local_layers.ptr;
126 
127  // detect errors in the kernel
128  auto k_error_propagate = this->error_propagate.device_ptr();
129  const auto k_npart_local = npart_local;
130 
131  this->sycl_target->queue
132  .submit([&](sycl::handler &cgh) {
133  cgh.parallel_for<>(
134  sycl::range<1>(pl_iter_range), [=](sycl::id<1> idx) {
135  NESO_PARTICLES_KERNEL_START
136  const INT cellx = NESO_PARTICLES_KERNEL_CELL;
137  const INT layerx = NESO_PARTICLES_KERNEL_LAYER;
138  // Inspect to see if the a mpi rank has been identified for a
139  // local communication pattern.
140  const auto mpi_rank_on_dat = k_mpi_rank_dat[cellx][1][layerx];
141  if (mpi_rank_on_dat < 0) {
142  // Atomically increment the lookup count
143  sycl::atomic_ref<int, sycl::memory_order::relaxed,
144  sycl::memory_scope::device>
145  atomic_count(k_lookup_count[0]);
146  const int index = atomic_count.fetch_add(1);
147 
148  NESO_KERNEL_ASSERT((index >= 0) && (index < k_npart_local),
149  k_error_propagate);
150 
151  // store this particles location so that it can be
152  // directly accessed later
153  k_lookup_local_cells[index] = cellx;
154  k_lookup_local_layers[index] = layerx;
155  }
156  NESO_PARTICLES_KERNEL_END
157  });
158  })
159  .wait_and_throw();
160  this->error_propagate.check_and_throw("Bad atomic index computed.");
161 
162  this->sycl_target->queue
163  .memcpy(this->h_lookup_count.ptr, this->d_lookup_count.ptr,
164  this->d_lookup_count.size_bytes())
165  .wait();
166  const auto npart_query = this->h_lookup_count.ptr[0];
167 
168  // these global indices are passed to the mesh hierarchy toget the rank
169  this->d_lookup_global_cells.realloc_no_copy(npart_query * 6);
170  this->h_lookup_global_cells.realloc_no_copy(npart_query * 6);
171  this->h_lookup_ranks.realloc_no_copy(npart_query);
172  this->d_lookup_ranks.realloc_no_copy(npart_query);
173 
174  auto k_lookup_global_cells = this->d_lookup_global_cells.ptr;
175  auto k_lookup_ranks = this->d_lookup_ranks.ptr;
176 
177  // variables required in the kernel to map positions to cells
178  auto mesh_hierarchy = this->h_mesh->get_mesh_hierarchy();
179  auto k_ndim = mesh_hierarchy->ndim;
180  const REAL k_inverse_cell_width_coarse =
181  mesh_hierarchy->inverse_cell_width_coarse;
182  const REAL k_inverse_cell_width_fine =
183  mesh_hierarchy->inverse_cell_width_fine;
184  const REAL k_cell_width_coarse = mesh_hierarchy->cell_width_coarse;
185  const REAL k_cell_width_fine = mesh_hierarchy->cell_width_fine;
186  const REAL k_ncells_dim_fine = mesh_hierarchy->ncells_dim_fine;
187 
188  for (int dimx = 0; dimx < k_ndim; dimx++) {
189  this->h_origin.ptr[dimx] = mesh_hierarchy->origin[dimx];
190  this->h_dims.ptr[dimx] = mesh_hierarchy->dims[dimx];
191  }
192  this->sycl_target->queue
193  .memcpy(this->d_origin.ptr, this->h_origin.ptr,
194  this->h_origin.size_bytes())
195  .wait();
196  this->sycl_target->queue
197  .memcpy(this->d_dims.ptr, this->h_dims.ptr, this->h_dims.size_bytes())
198  .wait();
199 
200  auto k_origin = this->d_origin.ptr;
201  auto k_dims = this->d_dims.ptr;
202 
203  // map particles positions to coarse and fine cells in the mesh hierarchy
204  this->sycl_target->queue
205  .submit([&](sycl::handler &cgh) {
206  cgh.parallel_for<>(sycl::range<1>(npart_query), [=](sycl::id<1> idx) {
207  const INT cellx = k_lookup_local_cells[idx];
208  const INT layerx = k_lookup_local_layers[idx];
209 
210  for (int dimx = 0; dimx < k_ndim; dimx++) {
211  // position relative to the mesh origin
212  const REAL pos =
213  k_position_dat[cellx][dimx][layerx] - k_origin[dimx];
214  const REAL tol = 1.0e-10;
215 
216  // coarse grid index
217  INT cell_coarse = ((REAL)pos * k_inverse_cell_width_coarse);
218  // bounds check the cell at the upper extent
219  if (cell_coarse >= k_dims[dimx]) {
220  // if the particle is within a given tolerance assume the
221  // out of bounds is a floating point issue.
222  if ((ABS(pos - k_dims[dimx] * k_cell_width_coarse) /
223  ABS(pos)) <= tol) {
224  cell_coarse = k_dims[dimx] - 1;
225  k_lookup_global_cells[(idx * k_ndim * 2) + dimx] =
226  cell_coarse;
227  } else {
228  cell_coarse = 0;
229  k_lookup_global_cells[(idx * k_ndim * 2) + dimx] = -2;
230  }
231  } else {
232  k_lookup_global_cells[(idx * k_ndim * 2) + dimx] = cell_coarse;
233  }
234 
235  // use the coarse cell index to offset the origin and compute
236  // the fine cell index
237  const REAL pos_fine = pos - cell_coarse * k_cell_width_coarse;
238  INT cell_fine = ((REAL)pos_fine * k_inverse_cell_width_fine);
239 
240  if (cell_fine >= k_ncells_dim_fine) {
241  if ((ABS(pos_fine - k_ncells_dim_fine * k_cell_width_fine) /
242  ABS(pos_fine)) <= tol) {
243  cell_fine = k_ncells_dim_fine - 1;
244  k_lookup_global_cells[(idx * k_ndim * 2) + dimx + k_ndim] =
245  cell_fine;
246  } else {
247  k_lookup_global_cells[(idx * k_ndim * 2) + dimx + k_ndim] =
248  -2;
249  }
250  } else {
251  k_lookup_global_cells[(idx * k_ndim * 2) + dimx + k_ndim] =
252  cell_fine;
253  }
254  }
255  });
256  })
257  .wait_and_throw();
258 
259  // copy the computed indicies to the host
260  if (this->d_lookup_global_cells.size_bytes() > 0) {
261  this->sycl_target->queue
262  .memcpy(this->h_lookup_global_cells.ptr,
263  this->d_lookup_global_cells.ptr,
264  this->d_lookup_global_cells.size_bytes())
265  .wait_and_throw();
266  }
267  // get the mpi ranks
268  mesh_hierarchy->get_owners(npart_query, this->h_lookup_global_cells.ptr,
269  this->h_lookup_ranks.ptr);
270 
271  // copy the mpi ranks back to the ParticleDat
272  if (this->h_lookup_ranks.size_bytes() > 0) {
273  this->sycl_target->queue
274  .memcpy(this->d_lookup_ranks.ptr, this->h_lookup_ranks.ptr,
275  this->h_lookup_ranks.size_bytes())
276  .wait_and_throw();
277  }
278 
279  this->sycl_target->queue
280  .submit([&](sycl::handler &cgh) {
281  cgh.parallel_for<>(sycl::range<1>(npart_query), [=](sycl::id<1> idx) {
282  const INT cellx = k_lookup_local_cells[idx];
283  const INT layerx = k_lookup_local_layers[idx];
284  k_mpi_rank_dat[cellx][0][layerx] = k_lookup_ranks[idx];
285  });
286  })
287  .wait_and_throw();
288

290  profile_elapsed(t0, profile_timestamp()));
291  };
