409  particle_data.contains(dat.first), cellids,
410  layers, particle_data.get(dat.first));
411  }

414  realloc_dat(dat.second);
415  dat.second->append_particle_data(npart_new,
416  particle_data.contains(dat.first), cellids,
417  layers, particle_data.get(dat.first));
418  }

423  dat.second->npart_host_to_device();

447 inline void ParticleGroup::remove_particles(const int npart,
448  const std::vector<INT> &cells,
449  const std::vector<INT> &layers) {
450 
451  this->d_remove_cells.realloc_no_copy(npart);
452  this->d_remove_layers.realloc_no_copy(npart);
453 
454  auto k_cells = this->d_remove_cells.ptr;
455  auto k_layers = this->d_remove_layers.ptr;
456 
457  NESOASSERT(cells.size() >= npart, "Bad cells length compared to npart");
458  NESOASSERT(layers.size() >= npart, "Bad layers length compared to npart");
459 
460  auto b_cells = sycl::buffer<INT>(cells.data(), sycl::range<1>(npart));
461  auto b_layers = sycl::buffer<INT>(layers.data(), sycl::range<1>(npart));
462 
463  this->sycl_target->queue
464  .submit([&](sycl::handler &cgh) {
465  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
466  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
467  cgh.parallel_for<>(sycl::range<1>(static_cast<size_t>(npart)),
468  [=](sycl::id<1> idx) {
469  k_cells[idx] = static_cast<INT>(a_cells[idx]);
470  k_layers[idx] = static_cast<INT>(a_layers[idx]);
471  });
472  })
473  .wait_and_throw();
474 
475  this->remove_particles(npart, this->d_remove_cells.ptr,
476  this->d_remove_layers.ptr);
477 }
