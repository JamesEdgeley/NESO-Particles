138  for (int cy = starts[1]; cy < ends[1]; cy++) {
139  const int cys = (cy + stencil_width * cell_counts[1]) % cell_counts[1];
140  index_mesh[1] = cys;
141  const int local_tuple_y = (this->ndim > 1) ? h_lookup_y[cys] : 0;
142  for (int cx = starts[0]; cx < ends[0]; cx++) {
143  const int cxs =
144  (cx + stencil_width * cell_counts[0]) % cell_counts[0];
145  index_mesh[0] = cxs;
146

149  this->mesh->mesh_tuple_to_mh_tuple(index_mesh, index_mh);

154  index_mh);

158 
159  const int linear_index =
160  local_tuple_x +
161  stride_x * (local_tuple_y + stride_y * local_tuple_z);

210  int mask = 0;
211  // k_mpi_rank_dat[cellx][1][layerx];
212  for (int dimx = 0; dimx < k_ndim; dimx++) {
213  const REAL pos = k_position_dat[cellx][dimx][layerx];

216  cell_fine = k_dims[dimx] - 1;

218  cell_fine = 0;
219  }
220  // use the cell to get an index for the dimension
221  const int dim_index =
222  k_lookup[k_lookup_stride * dimx + cell_fine];
223  local_tuple[dimx] = dim_index;
224

226  // negative value
227  // => cell is not in the local map
228  mask = MIN(mask, dim_index);
229  }
230 
231  const int linear_index =
232  local_tuple[0] +
233  k_lookup_dims[0] *
234  (local_tuple[1] + k_lookup_dims[1] * local_tuple[2]);
235 
236  const int remote_rank = (mask < 0) ? -1 : k_map[linear_index];
