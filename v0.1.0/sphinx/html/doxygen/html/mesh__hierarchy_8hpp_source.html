43  MPICHK(MPI_Win_fence(0, this->map_win))
44  if (comm_pair.rank_intra == 0) {
45  int *tmp_map = (int *)std::malloc(this->ncells_global * sizeof(int));

306  inline void claim_cell(const INT index, int weight) {
307  NESOASSERT((weight != mask) && (weight > 0) && (weight < (-1 * (mask + 1))),
308  "Invalid weight");

325  inline void claim_finalise() {
326 
327  MPICHK(MPI_Win_fence(0, this->map_win));
328  MPICHK(MPI_Barrier(this->comm_pair.comm_intra));
329 
330  // reduce accross all shared regions (inter node)
331  all_reduce_max_map();
332 
333  NESOASSERT(claim_stack_binned.empty(), "Claim stack is not empty.");
334 
335  while (!claim_stack.empty()) {
336  auto claim = claim_stack.top();
337 
338  // if the weight of this claim is in the cell then this rank is one of
339  // the ranks with the maximum weight.
340  if (claim.second >= map[claim.first]) {
341  claim_stack_binned.push(claim);
342  }
343  claim_stack.pop();
344  }
345 
346  MPICHK(MPI_Barrier(this->comm_pair.comm_intra));
347 
348  // for each rank with the maximum claimed weight reduce the ranks
349  const int rank = comm_pair.rank_parent;
350 
351  MPICHK(MPI_Win_fence(0, this->map_win));
352 
353  while (!claim_stack_binned.empty()) {
354  auto claim = claim_stack_binned.top();
355  INT index = claim.first;
356  int result;
357 
358  MPI_Win_lock(MPI_LOCK_SHARED, 0, 0, map_win);
359  MPICHK(MPI_Get_accumulate(&rank, 1, MPI_INT, &result, 1, MPI_INT, 0,
360  (MPI_Aint)index, 1, MPI_INT, MPI_MAX, map_win));
361  MPI_Win_unlock(0, map_win);
362  claim_stack_binned.pop();
363  }
364 
365  MPICHK(MPI_Win_fence(0, this->map_win));
366 
367  MPICHK(MPI_Barrier(this->comm_pair.comm_intra));
368  // reduce accross all shared regions (inter node)
369  all_reduce_max_map();
370 
371  // this->map should now contain either values of "mask" for cells that were
372  // not claimed and the rank of claimed cells.
373  map_created = true;
374  }
