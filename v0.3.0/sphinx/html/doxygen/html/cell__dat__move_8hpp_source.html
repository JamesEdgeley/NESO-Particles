181  inline void move() {
182  auto t0 = profile_timestamp();
183  // reset the particle counters on each cell
184  auto mpi_rank_dat = particle_dats_int[Sym<INT>("NESO_MPI_RANK")];
185  const auto k_ncell = this->ncell;
186  auto k_npart_cell = d_npart_cell.ptr;
187  auto k_mpi_npart_cell = mpi_rank_dat->d_npart_cell;
188  auto reset_event = this->sycl_target->queue.submit([&](sycl::handler &cgh) {
189  cgh.parallel_for<>(sycl::range<1>(k_ncell), [=](sycl::id<1> idx) {
190  k_npart_cell[idx] = k_mpi_npart_cell[idx];
191  });
192  });
193  auto k_move_count = d_move_count.ptr;
194  this->sycl_target->queue
195  .submit([&](sycl::handler &cgh) {
196  cgh.single_task<>([=]() { k_move_count[0] = 0; });
197  })
198  .wait_and_throw();
199  reset_event.wait_and_throw();
200 
201  // space to store particles moving between cells
202  const INT npart_local = mpi_rank_dat->get_npart_local();
203  this->d_cells_old.realloc_no_copy(npart_local);
204  this->d_cells_new.realloc_no_copy(npart_local);
205  this->d_layers_old.realloc_no_copy(npart_local);
206  this->d_layers_new.realloc_no_copy(npart_local);
207 
208  auto k_cells_old = this->d_cells_old.ptr;
209  auto k_cells_new = this->d_cells_new.ptr;
210  auto k_layers_old = this->d_layers_old.ptr;
211  auto k_layers_new = this->d_layers_new.ptr;
212 
213  // loop over particles and identify the particles to be move between
214  // cells.
215  auto pl_iter_range = mpi_rank_dat->get_particle_loop_iter_range();
216  auto pl_stride = mpi_rank_dat->get_particle_loop_cell_stride();
217  auto pl_npart_cell = mpi_rank_dat->get_particle_loop_npart_cell();
218  auto k_cell_id_dat = this->cell_id_dat->cell_dat.device_ptr();
219 
220  // detect out of bounds particles
221  auto k_ep_indices = this->ep_bad_cell_indices.device_ptr();
222 
223  this->sycl_target->queue
224  .submit([&](sycl::handler &cgh) {
225  cgh.parallel_for<>(
226  sycl::range<1>(pl_iter_range), [=](sycl::id<1> idx) {
227  NESO_PARTICLES_KERNEL_START
228  const INT cellx = NESO_PARTICLES_KERNEL_CELL;
229  const INT layerx = NESO_PARTICLES_KERNEL_LAYER;
230 
231  // if the cell on the particle is not the current cell then
232  // the particle needs moving.
233  const auto cell_on_dat = k_cell_id_dat[cellx][0][layerx];
234 
235  const bool valid_cell =
236  (cell_on_dat >= 0) && (cell_on_dat < k_ncell);
237  NESO_KERNEL_ASSERT(valid_cell, k_ep_indices);
238 
239  if ((cellx != cell_on_dat) && valid_cell) {
240  // Atomically increment the particle count for the new
241  // cell
242  sycl::atomic_ref<int, sycl::memory_order::relaxed,
243  sycl::memory_scope::device>
244  atomic_layer(k_npart_cell[cell_on_dat]);
245  const int layer_new = atomic_layer.fetch_add(1);
246 
247  // Get an index for this particle in the arrays that hold
248  // old/new cells/layers
249  sycl::atomic_ref<int, sycl::memory_order::relaxed,
250  sycl::memory_scope::device>
251  atomic_index(k_move_count[0]);
252  const int array_index = atomic_index.fetch_add(1);
253 
254  k_cells_old[array_index] = cellx;
255  k_cells_new[array_index] = static_cast<int>(cell_on_dat);
256  k_layers_old[array_index] = layerx;
257  k_layers_new[array_index] = layer_new;
258  }
259  NESO_PARTICLES_KERNEL_END
260  });
261  })
262  .wait_and_throw();
263  this->ep_bad_cell_indices.check_and_throw(
264  "Particle held bad cell id (not in [0,..,N_cell - 1]).");
265 
266  // Realloc the ParticleDat cells for the move
267  if (this->ncell > 0) {
268  this->sycl_target->queue
269  .memcpy(this->h_npart_cell.ptr, this->d_npart_cell.ptr,
270  sizeof(int) * this->ncell)
271  .wait();
272  }
273 
274  for (auto &dat : particle_dats_real) {
275  dat.second->realloc(this->h_npart_cell);
276  }
277  for (auto &dat : particle_dats_int) {
278  dat.second->realloc(this->h_npart_cell);
279  }
280 
281  // wait for the reallocs
282  for (auto &dat : particle_dats_real) {
283  dat.second->wait_realloc();
284  }
285  for (auto &dat : particle_dats_int) {
286  dat.second->wait_realloc();
287  }
288  this->sycl_target->queue
289  .submit([&](sycl::handler &cgh) {
290  cgh.parallel_for<class dummy>(
291  sycl::range<1>(k_ncell),
292  [=](sycl::id<1> idx) { k_npart_cell[idx] = 0; });
293  })
294  .wait();
295 
296  EventStack tmp_stack;
297  for (auto &dat : particle_dats_real) {
298  tmp_stack.push(dat.second->async_set_npart_cells(this->h_npart_cell));
299  }
300  for (auto &dat : particle_dats_int) {
301  tmp_stack.push(dat.second->async_set_npart_cells(this->h_npart_cell));
302  }
303  tmp_stack.wait();
304 
305  // get the npart to move on the host
306  this->sycl_target->queue
307  .memcpy(this->h_move_count.ptr, this->d_move_count.ptr, sizeof(int))
308  .wait();
309  const int move_count = h_move_count.ptr[0];
310 
311  // get the pointers into the ParticleDats
312  this->get_particle_dat_info();
313 
314  const int k_num_dats_real = this->num_dats_real;
315  const int k_num_dats_int = this->num_dats_int;
316  const auto k_particle_dat_ptr_real = this->d_particle_dat_ptr_real.ptr;
317  const auto k_particle_dat_ptr_int = this->d_particle_dat_ptr_int.ptr;
318  const auto k_particle_dat_ncomp_real = this->d_particle_dat_ncomp_real.ptr;
319  const auto k_particle_dat_ncomp_int = this->d_particle_dat_ncomp_int.ptr;
320 
321  auto t1 = profile_timestamp();
322  // copy from old cells/layers to new cells/layers
323  this->sycl_target->queue
324  .submit([&](sycl::handler &cgh) {
325  cgh.parallel_for<>(sycl::range<1>(move_count), [=](sycl::id<1> idx) {
326  const auto cell_old = k_cells_old[idx];
327  const auto cell_new = k_cells_new[idx];
328  const auto layer_old = k_layers_old[idx];
329  const auto layer_new = k_layers_new[idx];
330 
331  // loop over the ParticleDats and copy the data
332  // for each real dat
333  for (int dx = 0; dx < k_num_dats_real; dx++) {
334  REAL ***dat_ptr = k_particle_dat_ptr_real[dx];
335  const int ncomp = k_particle_dat_ncomp_real[dx];
336  // for each component
337  for (int cx = 0; cx < ncomp; cx++) {
338  dat_ptr[cell_new][cx][layer_new] =
339  dat_ptr[cell_old][cx][layer_old];
340  }
341  }
342  // for each int dat
343  for (int dx = 0; dx < k_num_dats_int; dx++) {
344  INT ***dat_ptr = k_particle_dat_ptr_int[dx];
345  const int ncomp = k_particle_dat_ncomp_int[dx];
346  // for each component
347  for (int cx = 0; cx < ncomp; cx++) {
348  dat_ptr[cell_new][cx][layer_new] =
349  dat_ptr[cell_old][cx][layer_old];
350  }
351  }
352  });
353  })
354  .wait_and_throw();
355  sycl_target->profile_map.inc("CellMove", "cell_move", 1,
356  profile_elapsed(t1, profile_timestamp()));
357 
358  auto t2 = profile_timestamp();
359  // compress the data by removing the old rows
360  this->layer_compressor.remove_particles(move_count, this->d_cells_old.ptr,
361  this->d_layers_old.ptr);
362 
363  sycl_target->profile_map.inc("CellMove", "remove_particles", 1,
364  profile_elapsed(t2, profile_timestamp()));
365  sycl_target->profile_map.inc("CellMove", "move", 1,
366  profile_elapsed(t0, profile_timestamp()));
367  };
