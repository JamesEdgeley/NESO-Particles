108  auto k_send_counts_all_ranks = this->dh_send_counts_all_ranks.d_buffer.ptr;
109  auto k_num_ranks_send = this->dh_num_ranks_send.d_buffer.ptr;

119  k_send_ranks[idx] = 0;

132  // packing layer by incrementing the send count for the report rank and
133  // increment the counter for the number of remote ranks to send to
134  const INT INT_comm_size = static_cast<INT>(comm_size);
135  const INT INT_comm_rank = static_cast<INT>(comm_rank);

144  const INT layerx = NESO_PARTICLES_KERNEL_LAYER;
145  const INT owning_rank =

147 
148  // if rank is valid and not equal to this rank then this
149  // particle is being sent somewhere
150  if (((owning_rank >= 0) && (owning_rank < INT_comm_size)) &&
151  (owning_rank != INT_comm_rank)) {

153  // reuse the recv ranks array to avoid mallocing more space
154  sycl::atomic_ref<int, sycl::memory_order::relaxed,

157  const int pack_layer = pack_layer_atomic.fetch_add(1);
158 
159  // increment the counter for number of sent particles (all
160  // ranks)
161  sycl::atomic_ref<int, sycl::memory_order::relaxed,
162  sycl::memory_scope::device>
163  send_count_atomic(k_num_particle_send[0]);

165

167  k_pack_cells[send_index] = static_cast<int>(cellx);
168  k_pack_layers_src[send_index] = static_cast<int>(layerx);
169  k_pack_layers_dst[send_index] = pack_layer;
170 
171  // if the packing layer is zero then this is the first
172  // particle found sending to the remote rank -> increment
173  // the number of remote ranks and record this rank.
174  if ((pack_layer == 0) && (rank_component == 0)) {
175  sycl::atomic_ref<int, sycl::memory_order::relaxed,
176  sycl::memory_scope::device>
177  num_ranks_send_atomic(k_num_ranks_send[0]);
178  const int rank_index = num_ranks_send_atomic.fetch_add(1);
179  k_send_ranks[rank_index] = static_cast<int>(owning_rank);
180  k_send_rank_map[owning_rank] = rank_index;
181  }
182  }

184  });
