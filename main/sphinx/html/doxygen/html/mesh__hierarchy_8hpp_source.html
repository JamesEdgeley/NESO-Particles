440  inline void map_to_tuple(const REAL *position, INT *cell) const {
441  for (int dimx = 0; dimx < ndim; dimx++) {
442  // position relative to the mesh origin
443  const REAL pos = position[dimx] - origin[dimx];
444  const REAL tol = 1.0e-10;
445 
446  // coarse grid index
447  INT cell_coarse = ((REAL)pos * inverse_cell_width_coarse);
448  // bounds check the cell at the upper extent
449  if (cell_coarse >= dims[dimx]) {
450  // if the particle is within a given tolerance assume the
451  // out of bounds is a floating point issue.
452  if ((ABS(pos - dims[dimx] * cell_width_coarse) / ABS(pos)) <= tol) {
453  cell_coarse = dims[dimx] - 1;
454  cell[dimx] = cell_coarse;
455  } else {
456  cell_coarse = 0;
457  cell[dimx] = -2;
458  }
459  } else {
460  cell[dimx] = cell_coarse;
461  }
462 
463  // use the coarse cell index to offset the origin and compute
464  // the fine cell index
465  const REAL pos_fine = pos - cell_coarse * cell_width_coarse;
466  INT cell_fine = ((REAL)pos_fine * inverse_cell_width_fine);
467 
468  if (cell_fine >= ncells_dim_fine) {
469  if ((ABS(pos_fine - ncells_dim_fine * cell_width_fine) /
470  ABS(pos_fine)) <= tol) {
471  cell_fine = ncells_dim_fine - 1;
472  cell[dimx + ndim] = cell_fine;
473  } else {
474  cell[dimx + ndim] = -2;
475  }
476  } else {
477  cell[dimx + ndim] = cell_fine;
478  }
