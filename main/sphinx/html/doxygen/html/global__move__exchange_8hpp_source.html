165  inline void npart_exchange_finalise() {
166 
167  auto t0_npart = profile_timestamp();
168  // Wait for the accumulation that counts how many remote ranks will send to
169  // this rank.
170  MPICHK(MPI_Wait(&this->mpi_request, MPI_STATUS_IGNORE));
171  this->num_remote_recv_ranks = this->recv_win_data[0];
172 
173  // send particle counts for ranks this ranks will send to and recv counts
174  // from the num_remote_ranks this rank will recv from.
175  this->h_recv_rank_npart.realloc_no_copy(this->num_remote_recv_ranks);
176  this->h_recv_requests.realloc_no_copy(this->num_remote_recv_ranks);
177  this->h_send_requests.realloc_no_copy(this->num_remote_send_ranks);
178 
179  // non-blocking recv of particle counts
180  for (int rankx = 0; rankx < this->num_remote_recv_ranks; rankx++) {
181  MPICHK(MPI_Irecv(&this->h_recv_rank_npart.ptr[rankx], 1, MPI_INT,
182  MPI_ANY_SOURCE, 42, this->comm,
183  &this->h_recv_requests.ptr[rankx]));
184  }
185 
186  // non-blocking send of particle counts
187  for (int rankx = 0; rankx < this->num_remote_send_ranks; rankx++) {
188  MPICHK(MPI_Isend(&this->h_send_rank_npart.ptr[rankx], 1, MPI_INT,
189  this->h_send_ranks.ptr[rankx], 42, this->comm,
190  &this->h_send_requests.ptr[rankx]));
191  }
192 
193  // space to store the remote ranks
194  this->h_recv_ranks.realloc_no_copy(this->num_remote_recv_ranks);
195  this->h_recv_status.realloc_no_copy(
196  MAX(this->num_remote_send_ranks, this->num_remote_recv_ranks));
197  MPICHK(MPI_Waitall(this->num_remote_recv_ranks, this->h_recv_requests.ptr,
198  this->h_recv_status.ptr));
199  // the order of the recvs gives an order to the ranks that will send to
200  // this rank
201  for (int rankx = 0; rankx < this->num_remote_recv_ranks; rankx++) {
202  const int remote_rank = this->h_recv_status.ptr[rankx].MPI_SOURCE;
203  this->h_recv_ranks.ptr[rankx] = remote_rank;
204  NESOASSERT(((remote_rank >= 0) &&

206  "Recv rank is invalid.");

208  "A remote rank is trying to send 0 (or fewer) particles to "

210  }

213  this->h_recv_status.ptr));
