532 template <typename T>
533 inline void ParticleDatT<T>::append_particle_data(const int npart_new,
534  const bool new_data_exists,
535  std::vector<INT> &cells,
536  std::vector<INT> &layers,
537  std::vector<T> &data) {
538  this->write_callback_wrapper(0);
539 
540  NESOASSERT(npart_new <= cells.size(), "incorrect number of cells");
541 
542  // using "this" in the kernel causes segfaults on the device so we make a
543  // copy here.
544  const size_t size_npart_new = static_cast<size_t>(npart_new);
545  const int ncomp = this->ncomp;
546  T ***d_cell_dat_ptr = this->impl_get();
547 
548  sycl::buffer<INT, 1> b_cells(cells.data(), sycl::range<1>{size_npart_new});
549  sycl::buffer<INT, 1> b_layers(layers.data(), sycl::range<1>{size_npart_new});
550 
551  // If data is supplied copy the data otherwise zero the components.
552  if (new_data_exists) {
553  sycl::buffer<T, 1> b_data(data.data(),
554  sycl::range<1>{size_npart_new * this->ncomp});
555  this->sycl_target->queue.submit([&](sycl::handler &cgh) {
556  // The cell counts on this dat
557  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
558  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
559  // The new data
560  auto a_data = b_data.template get_access<sycl::access::mode::read>(cgh);
561  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
562  const INT cellx = a_cells[idx];
563  const INT layerx = a_layers[idx];
564  // copy the data into the dat.
565  for (int cx = 0; cx < ncomp; cx++) {
566  d_cell_dat_ptr[cellx][cx][layerx] = a_data[cx * npart_new + idx];
567  }
568  });
569  });
570  } else {
571  this->sycl_target->queue.submit([&](sycl::handler &cgh) {
572  // The cell counts on this dat
573  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
574  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
575 
576  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
577  const INT cellx = a_cells[idx];
578  const INT layerx = a_layers[idx];
579  // zero the new components in the dat
580  for (int cx = 0; cx < ncomp; cx++) {
581  d_cell_dat_ptr[cellx][cx][layerx] = ((T)0);
582  }
583  });
584  });
585  }
586 
587  for (int px = 0; px < npart_new; px++) {
588  auto cellx = cells[px];
589  this->h_npart_cell[cellx]++;
590  }
591 }
