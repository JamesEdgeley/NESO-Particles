634  const std::vector<INT> &cells,
635  const std::vector<INT> &layers) {
636 
637  this->d_remove_cells.realloc_no_copy(npart);
638  this->d_remove_layers.realloc_no_copy(npart);
639 
640  auto k_cells = this->d_remove_cells.ptr;
641  auto k_layers = this->d_remove_layers.ptr;
642 
643  NESOASSERT(cells.size() >= npart, "Bad cells length compared to npart");
644  NESOASSERT(layers.size() >= npart, "Bad layers length compared to npart");
645 
646  auto b_cells = sycl::buffer<INT>(cells.data(), sycl::range<1>(npart));
647  auto b_layers = sycl::buffer<INT>(layers.data(), sycl::range<1>(npart));
648 
649  this->sycl_target->queue
650  .submit([&](sycl::handler &cgh) {
651  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
652  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
653  cgh.parallel_for<>(sycl::range<1>(static_cast<size_t>(npart)),
654  [=](sycl::id<1> idx) {
655  k_cells[idx] = static_cast<INT>(a_cells[idx]);
656  k_layers[idx] = static_cast<INT>(a_layers[idx]);
657  });
658  })
659  .wait_and_throw();
660 
661  this->remove_particles(npart, this->d_remove_cells.ptr,
