472 template <typename T>
473 inline void ParticleDatT<T>::append_particle_data(const int npart_new,
474  const bool new_data_exists,
475  std::vector<INT> &cells,
476  std::vector<INT> &layers,
477  std::vector<T> &data) {
478 
479  NESOASSERT(npart_new <= cells.size(), "incorrect number of cells");
480 
481  // using "this" in the kernel causes segfaults on the device so we make a
482  // copy here.
483  const size_t size_npart_new = static_cast<size_t>(npart_new);
484  const int ncomp = this->ncomp;
485  T ***d_cell_dat_ptr = this->cell_dat.device_ptr();
486 
487  sycl::buffer<INT, 1> b_cells(cells.data(), sycl::range<1>{size_npart_new});
488  sycl::buffer<INT, 1> b_layers(layers.data(), sycl::range<1>{size_npart_new});
489 
490  // If data is supplied copy the data otherwise zero the components.
491  if (new_data_exists) {
492  sycl::buffer<T, 1> b_data(data.data(),
493  sycl::range<1>{size_npart_new * this->ncomp});
494  this->sycl_target->queue.submit([&](sycl::handler &cgh) {
495  // The cell counts on this dat
496  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
497  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
498  // The new data
499  auto a_data = b_data.template get_access<sycl::access::mode::read>(cgh);
500  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
501  const INT cellx = a_cells[idx];
502  const INT layerx = a_layers[idx];
503  // copy the data into the dat.
504  for (int cx = 0; cx < ncomp; cx++) {
505  d_cell_dat_ptr[cellx][cx][layerx] = a_data[cx * npart_new + idx];
506  }
507  });
508  });
509  } else {
510  this->sycl_target->queue.submit([&](sycl::handler &cgh) {
511  // The cell counts on this dat
512  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
513  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
514 
515  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
516  const INT cellx = a_cells[idx];
517  const INT layerx = a_layers[idx];
518  // zero the new components in the dat
519  for (int cx = 0; cx < ncomp; cx++) {
520  d_cell_dat_ptr[cellx][cx][layerx] = ((T)0);
521  }
522  });
523  });
524  }
525 
526  for (int px = 0; px < npart_new; px++) {
527  auto cellx = cells[px];
528  this->h_npart_cell[cellx]++;
529  }
530 }
