610  * Append particle data to the ParticleDat. wait() must be called on the
611  * event_stack before use of the data. Assumes the dat has already been

614 template <typename T>
615 inline void ParticleDatT<T>::append_particle_data(
616  const int npart_new, const bool new_data_exists, std::vector<INT> &cells,
617  std::vector<INT> &layers, std::vector<T> &data, EventStack &es) {
618 
619  if (npart_new == 0) {
620  return;
621  }
622 
623  this->write_callback_wrapper(0);
624 
625  NESOASSERT(npart_new <= cells.size(), "incorrect number of cells");
626 
627  // using "this" in the kernel causes segfaults on the device so we make a
628  // copy here.
629  const size_t size_npart_new = static_cast<size_t>(npart_new);
630  const int ncomp = this->ncomp;
631  T ***d_cell_dat_ptr = this->impl_get();
632 
633  sycl::buffer<INT, 1> b_cells(cells.data(), sycl::range<1>{size_npart_new});
634  sycl::buffer<INT, 1> b_layers(layers.data(), sycl::range<1>{size_npart_new});
635 
636  // If data is supplied copy the data otherwise zero the components.
637  if (new_data_exists) {
638  sycl::buffer<T, 1> b_data(data.data(),
639  sycl::range<1>{size_npart_new * this->ncomp});
640  es.push(this->sycl_target->queue.submit([&](sycl::handler &cgh) {
641  // The cell counts on this dat
642  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
643  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
644  // The new data
645  auto a_data = b_data.template get_access<sycl::access::mode::read>(cgh);
646  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
647  const INT cellx = a_cells[idx];
648  const INT layerx = a_layers[idx];
649  // copy the data into the dat.
650  for (int cx = 0; cx < ncomp; cx++) {
651  d_cell_dat_ptr[cellx][cx][layerx] = a_data[cx * npart_new + idx];
652  }
653  });
654  }));
655  } else {
656  es.push(this->sycl_target->queue.submit([&](sycl::handler &cgh) {
657  // The cell counts on this dat
658  auto a_cells = b_cells.get_access<sycl::access::mode::read>(cgh);
659  auto a_layers = b_layers.get_access<sycl::access::mode::read>(cgh);
660 
661  cgh.parallel_for<>(sycl::range<1>(npart_new), [=](sycl::id<1> idx) {
662  const INT cellx = a_cells[idx];
663  const INT layerx = a_layers[idx];
664  // zero the new components in the dat
665  for (int cx = 0; cx < ncomp; cx++) {
666  d_cell_dat_ptr[cellx][cx][layerx] = ((T)0);
667  }
668  });
669  }));
670  }
671 
672  for (int px = 0; px < npart_new; px++) {
673  auto cellx = cells[px];
674  this->h_npart_cell[cellx]++;
675  }
676  es.push(this->async_npart_host_to_device());
677 }

679  * Append particle data to the ParticleDat. wait() must be called on the
680  * event_stack before use of the data. Assumes the dat has already been
681  * reallocated.
682  */
683 template <typename T>
684 inline void ParticleDatT<T>::append_particle_data(
685  ParticleDatSharedPtr<T> particle_dat,
686  const std::vector<INT> &h_npart_cell_existing,
687  const std::vector<INT> &h_npart_cell_to_add, EventStack &es) {
688 
689  if (particle_dat != nullptr) {
690  NESOASSERT(
691  this->ncell == particle_dat->ncell,
692  "Source and destination ParticleDats have different cell counts");
693  NESOASSERT(this->ncomp == particle_dat->ncomp,
694  "Source and destination ParticleDats have different number of "
695  "components");
696  }
697 
698  this->write_callback_wrapper(0);
699  for (int cellx = 0; cellx < this->ncell; cellx++) {
700  const INT layer_start = h_npart_cell_existing.at(cellx);
701  const INT layer_to_add = h_npart_cell_to_add.at(cellx);
702  if (layer_to_add > 0) {
703 
704  for (int colx = 0; colx < this->ncomp; colx++) {
705  auto d_ptr_dst = this->cell_dat.col_device_ptr(cellx, colx);
706  NESOASSERT(d_ptr_dst != nullptr, "Bad dst ptr.");
707  if (particle_dat == nullptr) {
708  es.push(this->sycl_target->queue.fill(
709  d_ptr_dst + layer_start, static_cast<T>(0), layer_to_add));
710  } else {
711  const auto d_ptr_src =
712  particle_dat->cell_dat.col_device_ptr(cellx, colx);
713  NESOASSERT(d_ptr_src != nullptr, "Bad src ptr.");
714  es.push(this->sycl_target->queue.memcpy(
