792  inline ParticleSetSharedPtr get_particles(std::vector<INT> &cells,
793  std::vector<INT> &layers) {
794  NESOASSERT(cells.size() == layers.size(),
795  "Cells and layers vectors have different sizes.");
796  const int num_particles = cells.size();
797  auto ps = std::make_shared<ParticleSet>(num_particles, this->particle_spec);
798  const INT num_cells = this->domain->mesh->get_cell_count();
799 
800  EventStack es;
801 
802  auto lambda_copy = [&](const std::size_t num_bytes_per_element, auto dat,
803  const INT particle_index, const INT cell,
804  const INT layer) {
805  const int ncomp = dat->ncomp;
806  for (int nx = 0; nx < ncomp; nx++) {
807  auto dest_ptr = ps->get_ptr(dat->sym, particle_index, nx);
808  const auto source_ptr = dat->col_device_const_ptr(cell, nx) + layer;
809  es.push(this->sycl_target->queue.memcpy(dest_ptr, source_ptr,
810  num_bytes_per_element));
811  }
812  };
813 
814  for (int px = 0; px < num_particles; px++) {
815  const INT cell = cells.at(px);
816  const INT layer = layers.at(px);
817  NESOASSERT((cell > -1) && (cell < num_cells), "Cell index not in range.");
818  NESOASSERT((layer > -1) && (layer < this->get_npart_cell(cell)),
819  "Layer index not in range.");
820  for (auto &dat : this->particle_dats_real) {
821  lambda_copy(sizeof(REAL), dat.second, px, cell, layer);
822  }
823  for (auto &dat : this->particle_dats_int) {
824  lambda_copy(sizeof(INT), dat.second, px, cell, layer);
825  }
826  }
827 
828  es.wait();
829  return ps;
