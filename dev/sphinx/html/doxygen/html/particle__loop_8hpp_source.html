443  inline void submit(const std::optional<int> cell = std::nullopt) {
444 
445  NESOASSERT(
446  (!this->loop_running) || (cell != std::nullopt),
447  "ParticleLoop::submit called - but the loop is already submitted.");
448  this->loop_running = true;
449  if (this->iteration_set_is_empty(cell)) {
450  return;
451  }
452 
453  auto t0 = profile_timestamp();
454 
455  auto global_info = this->create_global_info();
456  global_info.starting_cell = (cell == std::nullopt) ? 0 : cell.value();
457 
458  auto k_npart_cell_lb = this->d_npart_cell_lb;
459  auto is = this->iteration_set->get(cell);
460  auto k_kernel = this->kernel;
461 
462  const int nbin = std::get<0>(is);
463  this->sycl_target->profile_map.inc(
464  "ParticleLoop", "Init", 1, profile_elapsed(t0, profile_timestamp()));
465 
466  for (int binx = 0; binx < nbin; binx++) {
467  sycl::nd_range<2> ndr = std::get<1>(is).at(binx);
468  const size_t cell_offset = std::get<2>(is).at(binx);
469  this->event_stack.push(
470  this->sycl_target->queue.submit([&](sycl::handler &cgh) {
471  loop_parameter_type loop_args;
472  create_loop_args(cgh, loop_args, &global_info);
473  cgh.parallel_for<>(ndr, [=](sycl::nd_item<2> idx) {
474  const std::size_t index = idx.get_global_linear_id();
475  const size_t cellxs = idx.get_global_id(0) + cell_offset;
476  const size_t layerxs = idx.get_global_id(1);
477  const int cellx = static_cast<int>(cellxs);
478  const int layerx = static_cast<int>(layerxs);
479  ParticleLoopImplementation::ParticleLoopIteration iterationx;
480  if (layerx < k_npart_cell_lb[cellx]) {
481  iterationx.index = index;
482  iterationx.cellx = cellx;
483  iterationx.layerx = layerx;
484  iterationx.loop_layerx = layerx;
485  kernel_parameter_type kernel_args;
486  create_kernel_args(iterationx, loop_args, kernel_args);
487  Tuple::apply(k_kernel, kernel_args);
488  }
489  });
490  }));
491  }
492  }

498  inline void wait() {
499  NESOASSERT(this->loop_running,
500  "ParticleLoop::wait called - but the loop is not submitted.");
501  // wait for the loop execution to complete
502  this->event_stack.wait();
503  ParticleLoopImplementation::ParticleLoopGlobalInfo global_info =
504  this->create_global_info();
505  auto cast_wrapper = [&](auto t) { post_loop_cast(&global_info, t); };
506  auto post_loop_caller = [&](auto... as) { (cast_wrapper(as), ...); };
507  std::apply(post_loop_caller, this->args);
508  this->loop_running = false;
509  }
