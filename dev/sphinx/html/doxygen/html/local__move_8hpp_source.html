287  // find particles leaving through the local interface
288  this->departing_identify.identify(1);
289 
290  // reset the packer
291  this->particle_packer.reset();
292 
293  for (int rankx = 0; rankx < this->num_remote_send_ranks; rankx++) {
294  const int rank = this->h_send_ranks.ptr[rankx];
295  this->h_send_rank_npart.ptr[rankx] =
296  this->departing_identify.dh_send_counts_all_ranks.h_buffer.ptr[rank];
297  }
298 
299  // check no particles are trying to be send through the local method to a
300  // rank the local method has not setup communication patterns with
301  for (int rankx = 0; rankx < this->num_remote_send_ranks; rankx++) {
302  const int rank = this->h_send_ranks.ptr[rankx];
303  this->departing_identify.dh_send_counts_all_ranks.h_buffer.ptr[rank] = 0;
304  }
305  const int mpi_size = this->sycl_target->comm_pair.size_parent;
306  for (int rank = 0; rank < mpi_size; rank++) {
307  NESOASSERT(this->departing_identify.dh_send_counts_all_ranks.h_buffer
308  .ptr[rank] == 0,
309  "Particle is attempting to use local communication method for "
310  "an unreachable rank.");
311  }
312 
313  const int num_particles_leaving =
314  this->departing_identify.dh_num_particle_send.h_buffer.ptr[0];
315 
316  // start the packing
317  auto global_pack_event = this->particle_packer.pack(
318  this->num_remote_send_ranks, this->h_send_rank_npart,
319  this->dh_send_rank_map, num_particles_leaving,
320  this->departing_identify.d_pack_cells,
321  this->departing_identify.d_pack_layers_src,
322  this->departing_identify.d_pack_layers_dst, this->particle_dats_real,
323  this->particle_dats_int, 1);
324 
325  // exchange the send/recv counts with neighbours
326  this->npart_exchange_sendrecv();
327 
328  // allocate space to recv packed particles
329  particle_unpacker.reset(this->num_remote_recv_ranks,
330  this->h_recv_rank_npart, this->particle_dats_real,
331  this->particle_dats_int);
332 
333  // wait for the local particles to be packed.
334  global_pack_event.wait_and_throw();
335 
336  // send and recv packed particles from particle_packer.cell_dat to
337  // particle_unpacker.h_recv_buffer using h_recv_offsets.
338 
339  // start exchange particles
340  this->exchange_init();
341 
342  // remove the sent particles whilst the communication occurs
343  this->layer_compressor.remove_particles(
344  num_particles_leaving, this->departing_identify.d_pack_cells.ptr,
345  this->departing_identify.d_pack_layers_src.ptr);
346 
347  // finalise exchange particles
348  this->exchange_finalise();

351  this->particle_unpacker.unpack(particle_dats_real, particle_dats_int);
