320 inline void ParticleGroup::add_particles_local(
321  std::shared_ptr<ProductMatrix> product_matrix, const INT *d_cells,
322  const INT *d_layers, ParticleGroup *source_particle_group) {
323 
324  NESOASSERT(((d_layers == nullptr) && (d_cells == nullptr)) ||
325  ((d_layers != nullptr) && (d_cells != nullptr)),
326  "d_cells and d_layers must either both be nullptr or both be not "
327  "nullptr.");
328 
329  NESOASSERT(this->particle_spec.contains(product_matrix->spec->particle_spec),
330  "A particle property was passed to add_particles_local which does "
331  "not exist in the ParticleGroup.");
332 
333  const int num_products = product_matrix->num_products;
334  // reuse this space
335  this->d_remove_cells.realloc_no_copy(num_products);
336  this->d_remove_layers.realloc_no_copy(num_products);
337  INT *cells_ptr = this->d_remove_cells.ptr;
338  INT *layers_ptr = this->d_remove_layers.ptr;
339  auto d_pm = product_matrix->impl_get_const();
340 
341  // Either the cells are set in the product matrix or we need to default
342  // initialise them to 0.
343  const auto cell_id_sym = this->cell_id_dat->sym;
344  const int cell_id_index = product_matrix->spec->get_sym_index(cell_id_sym);
345 
346  // If cells not explicitly defined in the ProductMatrix.
347  if (cell_id_index == -1) {
348  if (d_cells == nullptr) { // If no parent cells use cell 0.
349  this->sycl_target->queue.fill(cells_ptr, (INT)0, num_products)
350  .wait_and_throw();
351  } else { // Copy parent cells.
352  this->sycl_target->queue
353  .memcpy(cells_ptr, d_cells, sizeof(INT) * num_products)
354  .wait_and_throw();
355  }
356  } else { // Use cells explicitly defined in the product matrix
357  INT const *source_cells_ptr = d_pm.ptr_int + num_products * cell_id_index;
358  this->sycl_target->queue
359  .memcpy(cells_ptr, source_cells_ptr, sizeof(INT) * num_products)
360  .wait_and_throw();
361  }
362 
363  // If any of the cells are -1 we trim those particles by setting layers to -1.
364  // Get the new cell occupancies at the destination layers of the particles to
365  // insert
366  // get dst layers also sets h_npart_cell, d_npart_cell
367  this->get_new_layers(num_products, cells_ptr, layers_ptr);
368  this->recompute_npart_cell_es();
369 
370  // Allocate space in all the dats for the new particles
371  this->realloc_all_dats();
372 
373  // copy the particle data into the particle dats
374  // for each particle property either there is data in the product matrix or
375  // it should be initalised to 0
376  EventStack es;
377 
378  // lambda which copies the property
379  auto lambda_copy = [&](auto dat, const auto index, const auto d_offsets,
380  const auto d_src) {
381  const int ncomp_local = dat->ncomp;
382  const int ncomp_matrix = product_matrix->spec->get_num_components(dat->sym);
383  NESOASSERT(ncomp_local == ncomp_matrix,
384  "Missmatch in the number of components in the product matrix "
385  "and the number of components on the particle group for a sym.");
386 
387  auto dat_ptr = dat->impl_get();
388  const int k_ncomp = dat->ncomp;
389  const int k_nproducts = num_products;
390  es.push(this->sycl_target->queue.submit([&](sycl::handler &cgh) {
391  cgh.parallel_for<>(
392  sycl::range<1>(static_cast<size_t>(num_products)),
393  [=](sycl::id<1> idx) {
394  const INT cell = cells_ptr[idx];
395  const INT layer = layers_ptr[idx];
396  // layer == -1 implies the entry is invalid somehow
397  if (layer > -1) {
398  for (int nx = 0; nx < k_ncomp; nx++) {
399  dat_ptr[cell][nx][layer] =
400  d_src[(d_offsets[index] + nx) * k_nproducts + idx];
401  }
402  }
403  });
404  }));
405  };
406 
407  // lambda to copy property from parent particle
408  auto lambda_parent_copy = [&](auto dat, auto dat_src) {
409  auto dat_ptr = dat->impl_get();
410  const auto dat_src_ptr = dat_src->impl_get_const();
411  const int k_ncomp = dat->ncomp;
412  es.push(this->sycl_target->queue.submit([&](sycl::handler &cgh) {
413  cgh.parallel_for<>(sycl::range<1>(static_cast<size_t>(num_products)),
414  [=](sycl::id<1> idx) {
415  const INT cell = cells_ptr[idx];
416  const INT layer = layers_ptr[idx];
417  const INT cell_src = d_cells[idx];
418  const INT layer_src = d_layers[idx];
419  // layer == -1 implies the entry is invalid somehow
420  if (layer > -1) {
421  for (int nx = 0; nx < k_ncomp; nx++) {
422  dat_ptr[cell][nx][layer] =
423  dat_src_ptr[cell_src][nx][layer_src];
424  }
425  }
426  });
427  }));
428  };
429 
430  // dispatch to the copy lambda or zero lambda depending on if the data exists
431  auto lambda_dispatch = [&](auto dat, const auto d_offsets, const auto d_src) {
432  const int index = product_matrix->spec->get_sym_index(dat->sym);
433  // if the property is not in the ProductMatrix
434  if (index == -1) {
435  if (d_cells == nullptr) { // Zero properties if no parents defined
436  zero_dat_properties(dat, num_products, cells_ptr, layers_ptr, es);
437  } else { // Else copy the property from the parent
438  // Is there a separate parent
439  if (source_particle_group != nullptr) {
440  // Does this dat exist in the separate parent
441  if (source_particle_group->contains_dat(dat->sym)) {
442  auto dat_src = source_particle_group->get_dat(dat->sym);
443  lambda_parent_copy(dat, dat_src);
444  } else {
445  NESOWARN(false, "Child ParticleGroup has a particle property which "
446  "does not exist on the parent ParticleGroup.");
447  zero_dat_properties(dat, num_products, cells_ptr, layers_ptr, es);
448  }
449  } else {
450  lambda_parent_copy(dat, dat);
451  }
452  }
453  } else { // Copy from the ProductMatrix
454  lambda_copy(dat, index, d_offsets, d_src);
455  }
456  };
457 
458  // launch the copies into the ParticleDats
459  for (auto &dat : this->particle_dats_real) {
460  lambda_dispatch(dat.second, d_pm.offsets_real, d_pm.ptr_real);
461  }
462  for (auto &dat : this->particle_dats_int) {
463  lambda_dispatch(dat.second, d_pm.offsets_int, d_pm.ptr_int);
464  }
465  // launch the copies of the npart cells into the ParticleDats
466  for (auto &dat : this->particle_dats_real) {
467  es.push(dat.second->async_set_npart_cells(this->h_npart_cell));
468  }
469  for (auto &dat : this->particle_dats_int) {
470  es.push(dat.second->async_set_npart_cells(this->h_npart_cell));
471  }
472 
473  // wait for the copy kernels
474  es.wait();
475  this->check_dats_and_group_agree();
476 }
